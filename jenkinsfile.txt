 pipeline {
    agent any
    
    environment {
       AWS_ACCESS_KEY_ID = credentials('AWS_ACCESS_KEY_ID')
       AWS_SECRET_ACCESS_KEY = credentials('AWS_SECRET_ACCESS_KEY')
    }

    parameters {
        choice(
            choices: ['apply', 'destroy'],
            description: 'Terraform action to apply or destroy',
            name: 'action'
            )
        
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

    stage('Terraform Init') {
            steps {
                script {
                    sh 'terraform init -get-plugins=false'
                }
            }
        }

        stage('Manually Install Plugins') {
            steps {
                script {
                    sh 'terraform providers schema -json > schema.json'
                    sh 'terraform providers mirror -source=terraform-providers/ -destination=.terraform/plugins'
                }
            }
        }

        stage('Terraform Plan') {
            steps {
                script {
                    sh 'terraform plan -input=false -lock=false -out=tfplan'
                }
            }
        }

        stage('Terraform Apply/Destroy') {
            steps {
                script {
                    if(params.action == 'apply'){
                        sh 'terraform apply -auto-approve tfplan'
                    } else {
                        sh 'terraform destroy -auto-approve'
                    }
                }
            }
        }
    }
}
